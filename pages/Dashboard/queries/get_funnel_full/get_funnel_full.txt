with base as (
  select
    l.id,
    l.status,
    o.expected_amount,
    o.estimated_commission,
    o.close_probability
  from leads l
  left join opportunities o on o.lead_id = l.id
),

etapas as (
  select
    count(*) filter (where status = 'nuevo') as nuevos,
    count(*) filter (where status = 'contactado') as contactados,
    count(*) filter (where status = 'calificado') as calificados,
    count(*) filter (where status = 'visita agendada') as visitas,
    count(*) filter (where status = 'negociacion') as negociacion,
    count(*) filter (where status = 'cerrado ganado') as ganados,

    -- 💰 dinero real
    sum(estimated_commission) filter (where status = 'cerrado ganado') as ingreso_real,

    -- 💰 dinero proyectado
    sum((expected_amount * close_probability)/100) filter (where status = 'negociacion') as ingreso_proyectado

  from base
),

conv as (
  select *,
    round(contactados * 100.0 / nullif(nuevos,0),1) as p1,
    round(calificados * 100.0 / nullif(contactados,0),1) as p2,
    round(visitas * 100.0 / nullif(calificados,0),1) as p3,
    round(negociacion * 100.0 / nullif(visitas,0),1) as p4,
    round(ganados * 100.0 / nullif(negociacion,0),1) as p5
  from etapas
),

diagnostico as (
  select *,
    case
      when p5 < 20 then '⚠️ Bajo cierre - revisar negociación'
      when p3 < 30 then '⚠️ Pocas visitas - problema comercial'
      when ingreso_real > 0 then '✅ Buen rendimiento'
      else 'Sin actividad relevante'
    end as mensaje
  from conv
)

select * from diagnostico;
